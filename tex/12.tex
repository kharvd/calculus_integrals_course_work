\section{Задание 12}

\begin{multline*}
    \int \frac{dx}{x \sqrt[3]{x-1}} =
    \left|\begin{array}{l}
        \sqrt[3]{x-1} = t \\
        dx = 3t^2 \, dt
    \end{array}\right| = 
    \int \frac{3t^2}{(t^3+1)t} \, dt = 3 \int \frac{t}{t^3+1} \, dt = \\ 
    = 3\left(\frac{1}{\sqrt{3}}\arctg\left(\frac{2t-1}{\sqrt{3}}\right) - 
    \frac{1}{3} \ln (1+t) + \frac{1}{6} \ln (1-t+t^2)\right),
\end{multline*}
где $ t = \sqrt[3]{x-1} $.
